datetime

datetime：处理日期和时间的标准库

from datetime import datetime   # datetime模块还包含datetime类，通过from datetime import datetime导入datetime这个类；如果仅导入import datetime，则必须引用全名datetime.datetime

now = datetime.now()      # 获取当前datetime，具体到秒后六位
today = datetime.today()  # 返回当前日期
print(now, type(now), today, type(now))
# 获取指定日期和时间
from datetime import datetime
dt = datetime(2015, 4, 19, 12, 20)   # 用指定日期时间创建 datetime
print(dt)

timestamp：1970年1月1日 00:00:00 UTC+00:00时区的时刻称为epoch time，记为0（1970年以前的时间timestamp为负数），当前时间就是相对于epoch time的秒数，称为timestamp；用timestamp数字表示时间

from datetime import datetime

datetime_now = datetime.now()
timestamp_now = datetime_now.timestamp()                        # 把datetime转换为timestamp，timestamp是一个浮点数，整数位表示秒
print(timestamp_now)
datetime_now = datetime.fromtimestamp(timestamp_now)            # 将timestamp转换成本地时间（北京时区、东8区，比标准时区时间早8小时）
datetime_now_UTC = datetime.utcfromtimestamp(timestamp_now)     # 将timestamp转换成UTC时间（格林威治标准时区时间）
print(datetime_now)
print(datetime_now_UTC)

1. 时间格式化

str转换：通常用户输入的日期和时间是字符串，需要通过 datetime.strptime() 把 str 转换为 datetime，再进行处理

from datetime import datetime

cday = datetime.strptime('2022-06-27 18:19:59', '%Y-%m-%d %H:%M:%S')      # 把str转换为datetime
print(cday)

now = datetime.now()
print(now.strftime('%Y-%m-%d %H:%M:%S'))                                  # 把datetime转换为str
格式化符号 描述
%y 两位数的年份表示（00-99）
%Y 四位数的年份表示（000-9999）
%m 月份（01-12）
%d 月内中的一天（0-31）
%H 24小时制小时数（0-23）
%I 12小时制小时数（01-12）
%M 分钟数（00-59）
%S 秒（00-59）
%a 本地简化星期名称
%A 本地完整星期名称
%b 本地简化的月份名称
%B 本地完整的月份名称
%c 本地相应的日期表示和时间表示
%j 年内的一天（001-366）
%p 本地A.M.或P.M.的等价符
%U 一年中的星期数（00-53）星期天为星期的开始
%w 星期（0-6），星期天为星期的开始
%W 一年中的星期数（00-53）星期一为星期的开始
%x 本地相应的日期表示
%X 本地相应的时间表示
%Z 当前时区的名称
%% %号本身

datetime加减：导入 timedelta 类，直接用 + 和 - 运算符

from datetime import datetime, timedelta

now = datetime.now()
print(now)
print(now + timedelta(hours=10))
print(now - timedelta(days=1))
print(now + timedelta(days=2, hours=12))

时区转换

from datetime import datetime, timedelta, timezone

tz_utc_8 = timezone(timedelta(hours=8))              # 创建时区UTC+8:00
now = datetime.now()
print(now)
dt = now.replace(tzinfo=tz_utc_8)                    # 强制设置为UTC+8:00，在时间后显示时区
print(dt)


utc_dt = datetime.utcnow().replace(tzinfo=timezone.utc)        # 拿到UTC时间，并强制设置时区为UTC+0:00
print(utc_dt)
bj_dt = utc_dt.astimezone(timezone(timedelta(hours=8)))        # astimezone()将转换时区为北京时间；通过astimezone()方法，可以转换到任意时区
print(bj_dt)
tokyo_dt = utc_dt.astimezone(timezone(timedelta(hours=9)))     # astimezone()将转换时区为东京时间
print(tokyo_dt)
bj_dt2 = tokyo_dt.astimezone(timezone(timedelta(hours=8)))     # astimezone()将tokyo_dt转换时区为北京时间
print(bj_dt2)
# 获取用户输入的日期和时间，以及一个时区信息，请编写一个函数将其转换为timestamp
import re
from datetime import datetime, timezone, timedelta
def to_timestamp(dt_str, tz_str):
    tz_timezone = int(re.split(r'[\+\:]+', tz_str)[1])              # 获取时区值
    dt_datetime = datetime.strptime(dt_str, '%Y-%m-%d %H:%M:%S')    # 将字符串转换为datetime类型
    dt_datetime_UTC = dt_datetime.replace(tzinfo=timezone(timedelta(hours=tz_timezone)))       # 转换为UTC时间
    dt_timestamp = dt_datetime_UTC.timestamp()                      # 转换为timestamp
    return dt_timestamp

dt_str, tz_str = input('Please input datetime and timezone: ')
# dt_str, tz_str = '2022-06-28 09:38:40', 'UTC+8:00'    # 1656380320.0

dt_result = to_timestamp(dt_str, tz_str)
print(dt_result)
